const shelf = document.getElementById('mediaShelf');
const modal = document.getElementById('modal');
const modalTitle = document.getElementById('modalTitle');
const modalMedia = document.getElementById('modalMedia');
const modalRating = document.getElementById('modalRating');
const modalReview = document.getElementById('modalReview');

mediaItems.forEach(item => {
  const div = document.createElement('div');
  div.className = 'media-item';
  div.innerHTML = `
    <img src="${item.cover}" class="media-cover ${item.style}" />
    <div class="media-title">${item.title}</div>
  `;
  div.onclick = () => openModal(item);
  shelf.appendChild(div);
});

function openModal(item) {
  modal.classList.add('show');
  modalTitle.textContent = item.title;
  modalMedia.innerHTML = '';
  modalRating.innerHTML = '‚òÖ'.repeat(item.rating) + '‚òÜ'.repeat(5 - item.rating);
  modalReview.textContent = item.review;

  if (item.type === 'video') {
    modalMedia.innerHTML = `<iframe src="${item.url}" height="360" allowfullscreen frameborder="0"></iframe>`;
  } else if (item.type === 'music') {
    modalMedia.innerHTML = `<audio controls src="${item.url}"></audio>`;
  } else if (item.type === 'ebook') {
    const viewer = document.createElement("div");
    viewer.id = "epub-viewer";
    viewer.style.height = "400px";
    modalMedia.appendChild(viewer);
    const book = ePub(item.url);
    book.renderTo("epub-viewer", { width: "100%", height: "100%" });
  }
}

function closeModal() {
  modal.classList.remove('show');
  modalMedia.innerHTML = '';
}
